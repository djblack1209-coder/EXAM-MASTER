# 检查备份数据的方法

## ✅ 正确的检查代码

### 方法 1：一次性检查（推荐）

```javascript
// 在控制台执行（一次性检查所有备份）
(function() {
  // 检查题库备份
  const bankBackup = wx.getStorageSync('v30_bank_backup');
  if (bankBackup) {
    try {
      const restored = JSON.parse(bankBackup);
      console.log('✅ 题库备份存在:', restored.length, '道题');
      console.log('备份数据预览:', restored.slice(0, 2)); // 显示前2道题
    } catch (e) {
      console.error('❌ 备份数据解析失败:', e);
    }
  } else {
    console.log('❌ 题库备份不存在');
  }
  
  // 检查文件备份
  const filesBackup = wx.getStorageSync('imported_files_backup');
  if (filesBackup) {
    try {
      const restored = JSON.parse(filesBackup);
      console.log('✅ 文件备份存在:', restored.length, '个文件');
      console.log('备份文件列表:', restored);
    } catch (e) {
      console.error('❌ 文件备份数据解析失败:', e);
    }
  } else {
    console.log('❌ 文件备份不存在');
  }
  
  // 检查清空前备份
  const clearBackup = wx.getStorageSync('v30_bank_backup_before_clear');
  if (clearBackup) {
    try {
      const restored = JSON.parse(clearBackup);
      console.log('✅ 清空前备份存在:', restored.length, '道题');
    } catch (e) {
      console.error('❌ 清空前备份数据解析失败:', e);
    }
  } else {
    console.log('ℹ️ 清空前备份不存在（正常，只有清空操作前才会创建）');
  }
})();
```

