### 测试 1：验证备份存在

```javascript
// 在控制台执行
(function() {
  const bankBackup = wx.getStorageSync('v30_bank_backup');
  const filesBackup = wx.getStorageSync('imported_files_backup');
  
  console.log('📊 备份状态:');
  console.log('- 题库备份:', bankBackup ? '✅ 存在' : '❌ 不存在');
  console.log('- 文件备份:', filesBackup ? '✅ 存在' : '❌ 不存在');
  
  if (bankBackup) {
    const restored = JSON.parse(bankBackup);
    console.log('- 题库备份数量:', restored.length, '道题');
  }
  
  if (filesBackup) {
    const restored = JSON.parse(filesBackup);
    console.log('- 文件备份数量:', restored.length, '个文件');
  }
})();
```

**预期结果**：
